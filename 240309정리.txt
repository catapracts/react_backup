const reducer = (count1, action) =>
{
    switch(action.type)
    {
        case 1 : 
         return count1 +1;
        
        case 10 : 
            return count1 +10;
        
        case 100 :

            <button onClick = { () => dispatch({type:10})}>+10</button>

            <button onClick = { () => dispatch({type:1000})}>+1000</button>
